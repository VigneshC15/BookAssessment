CREATE PROCEDURE [dbo].[GetBooksSortedByAuthor]
AS
BEGIN
    SELECT * FROM tbl_Books where IsActive=1 and IsDeleted=0  ORDER BY CONCAT(AuthorLastName, ' ', AuthorFirstName), Title
END
